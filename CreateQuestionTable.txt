CREATE TABLE `question` (
  `iQuestionID` int(11) NOT NULL AUTO_INCREMENT,
  `iPageID` int(11) NOT NULL,
  `vchName` varchar(100) NOT NULL,
  `iIndex` int(11) NOT NULL,
  `vchQuestionText` varchar(1000) DEFAULT NULL,
  `vchQuestionID` varchar(50) DEFAULT NULL,
  `vchQuestionType` varchar(45) DEFAULT NULL,
  `bRequired` bit(1) DEFAULT NULL,
  `vchMin` varchar(45) DEFAULT NULL,
  `vchMax` varchar(45) DEFAULT NULL,
  `bValidateText` bit(1) DEFAULT b'0',
  `vchAllowTypes` varchar(100) DEFAULT 'true, true, true, true',
  `vchValidSpecialCharacters` varchar(100) DEFAULT NULL,
  `vchDecimalPlaces` varchar(2) DEFAULT NULL,
  `vchValidationType` varchar(45) DEFAULT NULL,
  `vchAnswerChoices` varchar(10000) DEFAULT NULL,
  `vchOtherChoice` varchar(1000) DEFAULT NULL,
  `vchDisplayType` varchar(45) DEFAULT NULL,
  `vchNumberOfAnswers` varchar(10) DEFAULT NULL,
  PRIMARY KEY (`iQuestionID`),
  KEY `iPageID_idx` (`iPageID`),
  CONSTRAINT `iPageID` FOREIGN KEY (`iPageID`) REFERENCES `page` (`iPageID`) ON DELETE NO ACTION
) ENGINE=InnoDB AUTO_INCREMENT=64 DEFAULT CHARSET=latin1;
