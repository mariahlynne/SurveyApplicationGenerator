-- --------------------------------------------------------------------------------
-- Routine DDL
-- Note: comments before and after the routine body will not be stored by the server
-- --------------------------------------------------------------------------------
DELIMITER $$

CREATE DEFINER=`root`@`localhost` PROCEDURE `RenameApplication`(IN iApplicationID INT, IN vchTitle VARCHAR(255))
BEGIN
	UPDATE surveyapplication SET surveyapplication.vchTitle = vchTitle WHERE surveyapplication.iApplicationID = iApplicationID;
END
